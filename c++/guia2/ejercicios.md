# Ejercicio 1
```
#include <iostream>

int main() {
    int cantidad_unidades;
    float precio_unitario, importe_compra, importe_desc, importe_final;
    std::cout << "Ingrese el precio unitario del producto: ";
    std::cin >> precio_unitario;
    std::cout << "Ingrese la cantidad de unidades adquiridas: ";
    std::cin >> cantidad_unidades;
    importe_compra = precio_unitario * cantidad_unidades;
    importe_desc = importe_compra * 0.15;
    importe_final = importe_compra - importe_desc;
    std::cout << "El importe de la compra es: "<< importe_compra << std::endl;
    std::cout << "El importe del descuento es: "<< importe_desc << std::endl;
    std::cout << "El importe a pagar es: "<< importe_final << std::endl;
    return 0;
}
```
